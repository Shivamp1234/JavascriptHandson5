Question 6 : Find the Winner

Answer:

var Game_Winner = (S) => 
{
  countOfAditya=0;
  countOfDanish=0;
  for(i=0;i < S.length;i++){
    if(S[i]==="A"){
      countOfAditya++;
    }
    if(S[i]==="D"){
      countOfDanish++;
    }
  }
  if(countOfAditya > countOfDanish){
    return "Aditya";
  }
  else if(countOfAditya < countOfDanish){
    return "Danish";
  }
  else{
    return "Draw";
  }
}

{
  let reversedString="";

       reversedString=reversedString+str[i];
